# .github/workflows/deploy.yml
name: CI/CD Pipeline - Build, Test, and Deploy ML API

on:
  # Trigger the workflow on pushes to the main branch
  push:
    branches:
      - main

# Define the environment variables (secrets)
env:
  DOCKER_REGISTRY: docker.io
  DOCKER_IMAGE_NAME: yourusername/satellite-classifier # <-- REPLACE with your Docker Hub path
  DOCKER_TAG: ${{ github.sha }} # Use the commit SHA as a unique version tag

jobs:
  build_and_deploy:
    runs-on: ubuntu-latest
    
    steps:
    - name: 1. Checkout repository code
      uses: actions/checkout@v4

    # --- Build & Test ---
    - name: 2. Set up Python 3.10
      uses: actions/setup-python@v5
      with:
        python-version: "3.10"
        
    - name: 3. Install dependencies and run tests
      run: |
        pip install -r requirements.txt
        pip install pytest  # Install only for the tests
        pytest tests/
      # The job fails here if any test fails, preventing bad code from being deployed

    # --- Docker Build and Push ---
    - name: 4. Log in to Docker Hub
      uses: docker/login-action@v3
      with:
        username: ${{ secrets.DOCKER_USERNAME }}
        password: ${{ secrets.DOCKER_PASSWORD }}

    - name: 5. Build the Docker image
      uses: docker/build-push-action@v5
      with:
        context: .
        push: true
        tags: |
          ${{ env.DOCKER_REGISTRY }}/${{ env.DOCKER_IMAGE_NAME }}:latest
          ${{ env.DOCKER_REGISTRY }}/${{ env.DOCKER_IMAGE_NAME }}:${{ env.DOCKER_TAG }}
        cache-from: type=gha

    # --- Deployment (Example: Using a Cloud Service) ---
    # NOTE: Specific to your cloud provider (AWS/GCP/Azure).
    # This is a conceptual placeholder for the deployment tool command.
    - name: 6. Deploy to Cloud Run (Example Step)
      # This step requires setting up GCP credentials secrets (e.g., GCP_CREDENTIALS)
      # and using a specific deployment action (e.g., google-github-actions/deploy-cloudrun).
      run: echo "Deployment step goes here, e.g., using Cloud Run or App Runner."
